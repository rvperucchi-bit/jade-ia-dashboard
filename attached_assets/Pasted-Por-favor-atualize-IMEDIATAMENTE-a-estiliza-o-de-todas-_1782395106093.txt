// CÓDIGO DA BARRA DE DIGITAÇÃO PARA SUBSTITUIR NA TELA DE CHAT:

const [isRecording, setIsRecording] = useState(false); // Estado para ativar o modo gravação
const [inputText, setInputText] = useState('');

{/* RENDERIZAÇÃO DA BARRA DE BAIXO */}
<View style={styles.inputContainer}>
  <View style={styles.inputRowInside}>
    
    {!isRecording ? (
      <>
        {/* Botão de Anexo (+) Transparente */}
        <TouchableOpacity style={styles.innerBarButton} activeOpacity={0.6}>
          <Text style={styles.barIcon}>＋</Text>
        </TouchableOpacity>

        {/* Campo de Texto Limpo */}
        <TextInput 
          style={styles.textInputStyle}
          placeholder="Perguntar ao Sleek IA..."
          placeholderTextColor="#626880"
          value={inputText}
          onChangeText={setInputText}
          multiline
        />

        {/* Botão de Enviar ou Iniciar Gravação */}
        <TouchableOpacity 
          style={styles.sendButtonCircle} 
          activeOpacity={0.8}
          onPress={() => {
            if (inputText.trim().length === 0) {
              setIsRecording(true); // Se vazio, inicia gravação e mostra ondas
            } else {
              // Função de enviar texto
              setInputText('');
            }
          }}
        >
          <Text style={styles.sendIcon}>
            {inputText.trim().length > 0 ? '▲' : '🎙️'}
          </Text>
        </TouchableOpacity>
      </>
    ) : (
      /* MODO GRAVAÇÃO ATIVO: REMOVE O INPUT E EXIBE AS ONDAS SONORAS BRANCAS */
      <View style={styles.recordingWaveContainer}>
        <Text style={styles.recordingLabel}>Gravando áudio...</Text>
        
        {/* Ondas Sonoras Estilizadas Minimalistas em Branco */}
        <View style={styles.waveRow}>
          <View style={[styles.waveLine, { height: 12 }]} />
          <View style={[styles.waveLine, { height: 24 }]} />
          <View style={[styles.waveLine, { height: 18 }]} />
          <View style={[styles.waveLine, { height: 28 }]} />
          <View style={[styles.waveLine, { height: 14 }]} />
        </View>

        {/* Botão para parar a gravação e enviar */}
        <TouchableOpacity 
          style={[styles.sendButtonCircle, { backgroundColor: '#38A169' }]} 
          onPress={() => setIsRecording(false)}
          activeOpacity={0.8}
        >
          <Text style={[styles.sendIcon, { color: '#FFFFFF' }]}>✓</Text>
        </TouchableOpacity>
      </View>
    )}

  </View>
</View>
