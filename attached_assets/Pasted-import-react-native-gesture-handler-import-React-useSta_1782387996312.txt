import 'react-native-gesture-handler';

import { StyleSheet, Text, View, TextInput, TouchableOpacity, SafeAreaView, StatusBar, ScrollView, KeyboardAvoidingView, Platform } from 'react-native';
import { createDrawerNavigator, DrawerContentScrollView } from '@react-navigation/drawer';

// IMPORTAÇÃO DAS NOSSAS TELAS PREMIUM REALISTAS
import PipelineScreen from './screens/PipelineScreen';

// CONTROLE CUSTOMIZADO DA BARRA LATERAL (DRAWER)
function CustomDrawerContent(props) {
  const menuItems = [
    { label: '💬 Chat Central IA', route: 'Chat' },
    { label: '📊 Pipeline Comercial', route: 'Pipeline' },
    { label: '🔍 Prospecção Maps IA', route: 'Prospecting' },
    { label: '🌱 Carteira & Farmer', route: 'Farmer' },
  ];

  return (
    <DrawerContentScrollView {...props} style={styles.drawerContainer}>
      <View style={styles.drawerHeader}>
        <Text style={styles.drawerBrand}>Sleek CRM</Text>
        <Text style={styles.drawerUser}>Agente Autónomo Ativo</Text>
      </View>
      {menuItems.map((item, index) => (
        <TouchableOpacity key={index} style={styles.drawerItem} activeOpacity={0.7} onPress={() => props.navigation.navigate(item.route)}>
          <Text style={styles.drawerItemText}>{item.label}</Text>
        </TouchableOpacity>
      ))}
    </DrawerContentScrollView>
  );
}

// INTERFACE DE CHAT ESTILO CHATGPT / GEMINI
function ChatScreen({ navigation }) {
  const [inputText, setInputText] = useState('');

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="light-content" backgroundColor="#090A0F" />
      <View style={styles.topBar}>
        <TouchableOpacity style={styles.transparentButton} onPress={() => navigation.openDrawer()}>
          <Text style={styles.topIconText}>☰</Text> 
        </TouchableOpacity>
        <Text style={styles.topTitle}>Sleek IA</Text>
        <TouchableOpacity style={styles.transparentButton}><Text style={styles.topIconText}>⚙️</Text></TouchableOpacity>
      </View>

      <ScrollView contentContainerStyle={styles.chatArea} showsVerticalScrollIndicator={false}>
        <View style={styles.welcomeContainer}>
          <Text style={styles.aiLogoGlow}>🤖</Text>
          <Text style={styles.welcomeText}>Como posso alavancar suas vendas hoje?</Text>
        </View>
      </ScrollView>

      <KeyboardAvoidingView behavior={Platform.OS === 'ios' ? 'padding' : 'height'}>
        <View style={styles.inputContainer}>
          <View style={styles.inputRowInside}>
            <TouchableOpacity style={styles.innerBarButton}><Text style={styles.barIcon}>＋</Text></TouchableOpacity>
            <TextInput style={styles.textInputStyle} placeholder="Perguntar ao Sleek IA..." placeholderTextColor="#626880" value={inputText} onChangeText={setInputText} multiline />
            <TouchableOpacity style={styles.sendButtonCircle}>
              <Text style={styles.sendIcon}>{inputText.trim().length > 0 ? '▲' : '🎙️'}</Text>
            </TouchableOpacity>
          </View>
        </View>
      </KeyboardAvoidingView>
    </SafeAreaView>
  );
}

const Drawer = createDrawerNavigator();

export default function App() {
  return (
    <NavigationContainer>
      <Drawer.Navigator
        drawerContent={(props) => <CustomDrawerContent {...props} />}
        screenOptions={{
          headerShown: false,
          drawerStyle: { width: 280 },
          swipeEnabled: true,
        }}
      >
        <Drawer.Screen name="Chat" component={ChatScreen} />
        <Drawer.Screen name="Pipeline" component={PipelineScreen} />
        <Drawer.Screen name="Prospecting" component={ProspectingScreen} />
        <Drawer.Screen name="Farmer" component={FarmerScreen} />
      </Drawer.Navigator>
    </NavigationContainer>
  );
}
